CREATE Procedure [dbo].[Ekle_CariAdres] (
@CariId Bigint,
@AdresAd NVarChar(100)='',
@Adres NVarChar(200)='',
@Sehir NVarChar(50)='',
@Ilce NVarChar(50)='',
@Telefon NVarChar(15)=''
) As 
Begin
Declare @SehirId Bigint
DEclare @IlIlceId Bigint
Declare @IDTab Table (ID Bigint)
Declare @Varsayilan As Bit = 1

	If Exists(SELECT ID FROM CARI_ADRES WHERE CARI=@CariId and VARSAYILAN=1)
	Begin
		Set @Varsayilan = 0
	End

	if @Sehir='' 
	Begin
		Print 'Şehir Boş Olamaz'
		Return
	End
	if @Ilce='' 
	Begin
		Print 'İlçe Boş Olamaz'
		Return
	End
	if Exists(Select * From ILILCE Where AD=Upper(@Sehir)) 
		Select @SehirId=ID FROM ILILCE WHERE AD=Upper(@Sehir)
	Else
	Begin
		Print 'Şehir Bulunamadı '+ @Sehir
		Return
	End
	
	if Exists(Select * From ILILCE Where USTID=@SehirId And AD=Upper(@Ilce)) 
		Select @IlIlceId=ID From ILILCE WHERE USTID=@SehirId And AD=Upper(@Ilce)
	Else
	Begin
		Print 'İlçe Bulunamadı '+ @Ilce
		Return
	End

	Declare @AdresId Bigint
	
	DELETE FROM @IDTab
	Insert Into @IDTab(ID) EXEC SEQUENS_VER @TABLO = N'CARI_ADRES',	@KOD_PC = 1000000
	SELECT @AdresId=ID FROM @IDTab

	INSERT INTO CARI_ADRES (ID,CARI,AD,ADRES,ILILCE,TELEFON,VARSAYILAN,GUN)
	VALUES (@AdresId,@CariId,@AdresAd,@Adres,@IlIlceId,@Telefon,@Varsayilan,1)

	SELECT @CariId CariId,@AdresId AdresId

End
