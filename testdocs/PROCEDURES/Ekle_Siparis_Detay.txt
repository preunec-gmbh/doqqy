

CREATE Procedure [dbo].[Ekle_Siparis_Detay](
@SipId Bigint,
@Lokasyon Bigint,
@StokId Bigint,
@Barkod NVarChar(max)='',
@Miktar Decimal(13,3)=1,
@Fiyat Decimal(18,8)=0) As
Begin
	Declare @IDTab Table(ID Bigint)

	if Not Exists(Select * From SIPARIS Where ID=@SipId)
	Begin
		Print 'Siparis Bulunamadı'
		Return
	End
	if @StokId Is Null Or @StokId=0
	Begin
		Print 'Stok ID Girilmemiş'
		Return
	End
	if Not Exists(Select * From STOK Where ID=@StokId)
	Begin
		Print 'Belirtilen ID : ' + RTrim(Cast(@StokId as Char(20))) + ' Herhangi bir Stok Kartına Ait Değil'
		Return		
	End
	if Not Exists(Select B.* From STOK_BARKOD B LEFT JOIN STOK_STOK_BIRIM R ON R.ID=B.STOK_STOK_BIRIM Where R.STOK=@StokId And B.BARKOD=@Barkod)
	Begin
		Print 'Belirtilen Barkod, Belirtilen Stok Kartına Ait Değil. Barkod : ' + @Barkod
		Return
	End 

	Declare @SipDetId Bigint
	Declare @CinsiId Bigint
	Declare @BirimId Bigint
	Declare @KdvOran Decimal(18,8)
	Declare @Carpan Decimal(18,8)
	SELECT @CinsiId=S.STOK_CINSI,@KdvOran=V.KDV_PAREKENDE FROM STOK S LEFT JOIN STOK_VERGI V ON V.ID=S.STOK_VERGI WHERE S.ID=@StokId
	SELECT @BirimId=GF.STOK_BIRIM,@Carpan=GF.CARPAN FROM STOK_BARKOD BR LEFT JOIN STOK_STOK_BIRIM GF ON GF.ID=BR.STOK_STOK_BIRIM WHERE BARKOD=@Barkod
	if @BirimId Is Null
	SELECT TOP 1 @BirimId=STOK_BIRIM,@Carpan=CARPAN FROM STOK_STOK_BIRIM WHERE STOK=@StokId
	
	Delete From @IDTab
	Insert Into @IDTab(ID) EXEC SEQUENS_VER @TABLO = N'SIPARIS_DETAY',	@KOD_PC = 1000000
	SELECT @SipDetId=ID FROM @IDTab	

	Insert Into SIPARIS_DETAY (ID,SIPARIS,LOKASYON,STOK,STOK_CINSI,STOK_BIRIM,BARKOD,DOVIZ_AD,CARPAN,MIKTAR_FIS,MIKTAR_BEDELSIZ,ANLASMA_FIYAT,FIYAT,DAHIL_FIYAT,TUTAR,DAHIL_TUTAR,ISKONTO,ISKONTO_HESAP,OTV_ORAN,OTV_TUTAR,KDV_TOPTAN,TEVKIF,KUR,FIYAT_FARKI,SERINO_ZORUNLU,YEREL_KARSI_FIYAT,BELGE_TARIHINDEKI_SON_ALIS_FIYATI,TOPLAM_SATIR_ISKONTOSU,TOPLAM_FIS_ISKONTOSU,TOPLAM_OTV,TOPLAM_KDV_MATRAHI,TOPLAM_KDV,TOPLAM_TEVKIF,HESAPLANAN_FIYAT,ACIKLAMA,KOD)
	VALUES (@SipDetId,@SipId,@Lokasyon,@StokId,@CinsiId,@BirimId,@Barkod,1,@Carpan,@Miktar,0,0,@Fiyat,@Fiyat*((@KdvOran/100)+1),(@Miktar*@Fiyat),(@Miktar*@Fiyat)*((@KdvOran/100)+1),'0',0,0,0,@KdvOran,0,1,0,0,0,0,0,0,0,(@Miktar*@Fiyat),(@Miktar*@Fiyat)/100*@KdvOran,0,0,'','')

	Update SIPARIS Set 
		SATIR_TOPLAM=(SELECT SUM(TUTAR) FROM SIPARIS_DETAY WHERE SIPARIS=@SipId),
		KDV_TOPLAM=(SELECT SUM(TOPLAM_KDV) FROM SIPARIS_DETAY WHERE SIPARIS=@SipId),
		GENELTOPLAM=(SELECT SUM(TOPLAM_KDV_MATRAHI+TOPLAM_KDV) FROM SIPARIS_DETAY WHERE SIPARIS=@SipId),
		SATIR_ISKONTO_TOPLAM=0,
		SIPARIS_ISKONTO_ORAN=0,
		YUVARLAMA=0,
		OTV_TOPLAM=0,
		TEFKIFAT_TOPLAM=0
		WHERE ID=@SipId

	SELECT @SipId SIPARIS,@SipDetId SIPARIS_DETAY

	

End
