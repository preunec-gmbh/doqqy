CREATE Procedure [dbo].[Ekle_CariKart] (
@CariKod NVarChar(50),
@CariUnvan NVarChar(150),
@CariGrup NVarChar(50)='GENEL',
@CariTur NVarChar(50)='CARI',
@VergiDairesi NVarChar(50)='',
@VergiNo NVarChar(20)='',
@TCKimlikNo NVarChar(11)='',
@Web NVarChar(50)='',
@EMail NVarChar(50)='',
@Adres NVarChar(200)='',
@Sehir NVarChar(50)='',
@Ilce NVarChar(50)='',
@Telefon NVarChar(15)='',
@TCVarsaEkle Bit=False,
@TCYoksaEmailKontrolEt Bit=1) As 
Begin
Declare @SehirId Bigint
DEclare @IlIlceId Bigint
Declare @CariGrupId Bigint
Declare @CariTurId Bigint
Declare @YS Char(2)
Declare @StokFiyatAd Bigint
Declare @CariId Bigint
Declare @IDTab Table (ID Bigint)

Set @YS=Char(13)+Char(10)
Set @CariId=0
	Select TOP 1 @StokFiyatAd = STOK_FIYAT_AD_VARSAYILAN From FIRMASABITLERI 
	if @CariKod='' 
	Begin
		Print 'Cari Kod Boş Olamaz. Yeni Kod Üretmek İçin + Yazınız'
		return
	End
	if @CariKod='+'
	Begin
		Set @CariKod='CK'+REPLACE(REPLACE(REPLACE(CONVERT(VARCHAR(25), GETDATE(), 126),'-',''),':',''),'.','')
	End
	if @CariUnvan=''
	Begin
		Print 'Cari Ünvan Boş Olamaz'
		Return
	End
	if @TCKimlikNo Is Not Null And @TCKimlikNo<>'' And @TCVarsaEkle=0 And Exists(SELECT ID FROM CARI WHERE KIMLIK_NO=@TCKimlikNo And KIMLIK_NO<>'' And @TCKimlikNo<>'11111111111')
	Begin
		Print 'TCKimlik No Daha Önce Kullanılmış'
		Return
	End
	if @TCKimlikNo='' And (@VergiDairesi='' or @VergiNo='')
	Begin
		Print 'TCKimlikNo veya VergiDairesi ve VergiNo Boş Olamaz'
		Return
	End
	/* Cari Grup */
	if @CariGrup=''
	Begin
		Select Top 1 @CariGrup=AD,@CariGrupId=ID From CARI_GRUP ORDER BY ID ASC
	End
	if @CariGrup<>''
	Begin
		if EXISTS(Select * FROM CARI_GRUP WHERE AD=@CariGrup)
		SELECT @CariGrupId=ID From CARI_GRUP WHERE AD=@CariGrup Else 
		Begin
			Print 'Cari Grup Bulunamadı '+@CariGrup
			Return
		End
	End
	/* Cari Tür */
	if @CariTur=''
	Select TOP 1 @CariTur=AD,@CariTurId=ID From CARI_TUR ORDER BY ID ASC
	Else
	Begin
		if Exists(Select * From CARI_TUR WHERE AD=@CariTur)
		Select TOP 1 @CariTur=AD,@CariTurId=ID From CARI_TUR WHERE AD=@CariTur Else
		Begin
			Print 'Cari Tür Bulunamadı '+@CariTur
			Return
		End	
	End
	if @Sehir='' 
	Begin
		Print 'Şehir Boş Olamaz'
		Return
	End
	if @Ilce='' 
	Begin
		Print 'İlçe Boş Olamaz'
		Return
	End
	if Exists(Select * From ILILCE Where AD=Upper(@Sehir)) 
		Select @SehirId=ID FROM ILILCE WHERE AD=Upper(@Sehir)
	Else
	Begin
		Print 'Şehir Bulunamadı '+ @Sehir
		Return
	End
	
	if Exists(Select * From ILILCE Where USTID=@SehirId And AD=Upper(@Ilce)) 
		Select @IlIlceId=ID From ILILCE WHERE USTID=@SehirId And AD=Upper(@Ilce)
	Else
	Begin
		Print 'İlçe Bulunamadı '+ @Ilce
		Return
	End

	if @TCYoksaEmailKontrolEt=1 And Exists(Select TOP 1 C.ID  FROM CARI C WHERE Upper(C.EMAIL)=Upper(@EMail) AND UPPER(C.AD)=UPPER(@CariUnvan) And @TCKimlikNo<>'11111111111')
	Begin
		Print 'Cari Zaten Var Mail'
		Select TOP 1 @CariId=C.ID  FROM CARI C WHERE UPPER(C.EMAIL)=UPPER(@EMail) AND UPPER(C.AD)=UPPER(@CariUnvan)
		Print @CariId
	End
	if @CariId=0 And Exists(Select TOP 1 * From CARI Where @TCKimlikNo<>'' And @TCKimlikNo<>'11111111111' And  KIMLIK_NO=@TCKimlikNo) And @TCVarsaEkle=0
	Begin
		Print 'Cari Zaten Var Kimlik'
		SELECT TOP 1 @CariId=ID FROM CARI Where KIMLIK_NO=@TCKimlikNo
	End
	Else
	Begin
		if @CariId=0
		Begin
			DELETE FROM @IDTab
			Insert Into @IDTab(ID) EXEC SEQUENS_VER @TABLO = N'CARI',	@KOD_PC = 1000000
			SELECT @CariId=ID FROM @IDTab

			INSERT INTO CARI (ID,CARI_TUR,KOD,AD,CARI_GRUP,USTID,ISKONTO,VADE,VADE_SECENEKLERI,VADE_FARKI,RISK,RISK_DAVRANISI,STOK_FIYAT_AD,VERGI_NUMARASI,KIMLIK_NO,VERGI_DAIRESI,WEB,EMAIL,ALIS_YAPILIR,SATIS_YAPILIR,ODEME_YAPILIR,IADE_ALINIR,IADE_VERILIR,SIPARIS_ZORUNLU,DOVIZ_AD,CARI_TAKIP_SEKLI,TAKSIT_GUNU,TAKSIT_KESIM_GUNU,TAKSIT_SAYISI,AKTIF,TARIH,TEMSILCI,POS_EKRANIDA_GOZUKUR,ISKONTO_ALIS,AIDAT,E_FATURA_TIPI,EK_AD)
			VALUES (@CariId,@CariTurId,@CariKod,@CariUnvan,@CariGrupId,0,'',0,7,0,0,3,@StokFiyatAd,@VergiNo,@TCKimlikNo,@VergiDairesi,@Web,@EMail,1,1,1,1,1,0,1,1,0,0,0,1,GETDATE(),0,1,'',0,1,'')
		End
	End

	Declare @AdresId Bigint
	Declare @AdresTb Table (CariId Bigint,AdresId Bigint)

	Exec Ekle_CariAdres @CariId,'Web Adres',@Adres,@Sehir,@Ilce,@Telefon

	Print 'Başladı'+@YS+'Cari Kod : '+@CariKod + @YS + 'Cari Grup : ' + @CariGrup+ Cast(@CariGrupID As Char(20)) + @YS 
	+ 'Cari Tür : ' + @CariTur + Cast(@CariTurId as Char(20)) + @YS
	+ 'Cari ID : ' + Cast(@CariId as Char(20))

	UPDATE C 
	SET C.ISKONTO=CG.ISKONTO, C.VADE=CG.VADE, C.VADE_FARKI=CG.VADE_FARKI, C.RISK=CG.RISK, C.STOK_FIYAT_AD=CG.STOK_FIYAT_AD, C.VADE_SECENEKLERI=CG.VADE_SECENEKLERI,
		C.ALIS_YAPILIR=CG.ALIS_YAPILIR, C.SATIS_YAPILIR=CG.SATIS_YAPILIR, C.ODEME_YAPILIR=CG.ODEME_YAPILIR, C.IADE_ALINIR=CG.IADE_ALINIR, C.IADE_VERILIR=CG.IADE_VERILIR,
		C.SIPARIS_ZORUNLU=CG.SIPARIS_ZORUNLU, C.POS_EKRANIDA_GOZUKUR=CG.POS_EKRANIDA_GOZUKUR, C.URUN_PROMASYONUNDAN_YARALANIR=CG.URUN_PROMASYONUNDAN_YARALANIR, 
		C.TOPLAM_PROMASYONLARINDAN_YARALANIR=CG.TOPLAM_PROMASYONLARINDAN_YARALANIR, C.TAKSITLI_SATIS_CARISI=CG.TAKSITLI_SATIS_CARISI, C.TAKSIT_GUNU=CG.TAKSIT_GUNU, 
		C.TAKSIT_KESIM_GUNU=CG.TAKSIT_KESIM_GUNU, C.TAKSIT_SAYISI=CG.TAKSIT_SAYISI, C.MAX_TAKSIT_SAYISI=CG.MAX_TAKSIT_SAYISI, C.PESINAT_YUZDESI=CG.PESINAT_YUZDESI, 
		C.VERESIYE_VERILMEZ=CG.VERESIYE_VERILMEZ
	FROM CARI C
	INNER JOIN CARI_GRUP CG ON C.CARI_GRUP=CG.ID
	WHERE C.ID=@CariId AND CG.ID=@CariGrupId


End
